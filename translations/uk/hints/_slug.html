Останнє запитання. Як виглядатиме моє остаточне рішення? 
import datetime


def get_latest(datetime_a: str, datetime_b: str) -> str:
    datetime_b_obj = datetime.datetime.fromisoformat(datetime_a)
    time_b_str_obj = datetime.datetime.fromisoformat(datetime_b)

    latest = datetime_b_obj if datetime_b_obj > time_b_str_obj else time_b_str_obj
    result = latest.isoformat()

    return result
