import datetime


def get_latest(datetime_a: str, datetime_b: str) -> str:
    datetime_a_obj = datetime.datetime.fromisoformat(datetime_a)
    datetime_b_obj = datetime.datetime.fromisoformat(datetime_b)

    latest = datetime_b_obj if datetime_b_obj > datetime_a_obj else datetime_a_obj
    result = latest.isoformat()

    return result
